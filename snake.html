function update() { if (!gameStarted) return; const head = { x: snake[0].x + direction.x * boxSize, y: snake[0].y + direction.y * boxSize }; // Teleportation bei Kollision mit Wänden if (head.x < 0) head.x = worldSize - boxSize; // links if (head.x >= worldSize) head.x = 0; // rechts if (head.y < 0) head.y = worldSize - boxSize; // oben if (head.y >= worldSize) head.y = 0; // unten // Kollision mit Hindernissen if (obstacles.some(obstacle => obstacle.x === head.x && obstacle.y === head.y)) { alert('Game Over!'); clearInterval(interval); location.reload(); } // Kollision mit Essen if (head.x === food.x && head.y === food.y) { score++; scoreElement.innerText = score; food = generateRandomPosition(); } else { snake.pop(); } // Kollision mit Power-up if (head.x === powerUp.x && head.y === powerUp.y) { powerUp = generateRandomPosition(); for (let i = 0; i < 3; i++) { snake.push({ ...snake[snake.length - 1] }); } } // Kollision mit eigenem Körper if (snake.some(segment => segment.x === head.x && segment.y === head.y)) { alert('Game Over!'); clearInterval(interval); location.reload(); } snake.unshift(head); // Offset berechnen const offsetX = head.x - viewSize / 2; const offsetY = head.y - viewSize / 2; // Zeichne alles neu ctx.clearRect(0, 0, canvas.width, canvas.height); drawWorldBorder(); // Weltgrenze zeichnen drawSnake(offsetX, offsetY); drawFood(offsetX, offsetY); drawPowerUp(offsetX, offsetY); drawObstacles(offsetX, offsetY); drawMiniMap(); }
